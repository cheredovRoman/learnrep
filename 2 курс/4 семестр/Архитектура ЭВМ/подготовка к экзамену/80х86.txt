M:.db 1,2,3,4,5,6; массив М имеет 6 элементов(1,2,3,4,5,6)
x:.db 2; переменная х=2
NumA:.db 4;переменная а=4
counter:.db 0; счетчик=0
result:.db 1;результат=1
begin: load a,size; Метка begin,загружаем в регистр а, размер массива
cycl:push a;начала цикла(cycl:), а закинули в стек(здесь закинули размер массива)
load a,counter;В регистр а закинули счетчик
adi a,1; регистр а+1, тобишь увеличели счетчик
stor a,counter; сохранили число из регистра а в переменную счетчик
load b,NumA; загрузили в регистр б число а
call stepen; вызов подпрограммы на возведение степень
push b;В регистр б закинули а в степени
load b,counter; загрузили в б регистр счетчик
sbi b,1; уменьшили счетчик на 1
load a,M(b); загружаем в регистр а Число из массива по счету из регистра б 
pop b;вытащили в б регистр число из стека(а в степени)
sbr a,b;регистр а минус регистр б
mov b,a; из регистра а в регистр б
mum a,result;регистр б умножаем на результат
stor b,result; сохраняем регистр б в результат
pop a; востанавливаем в регистр а размер массива
loop a,cycl;замыкание по метке цикл
